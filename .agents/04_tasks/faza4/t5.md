### 3.2 KPI kartice
Podaci se dobijaju sa `GET /api/v{version}/dashboard/kpi`

```typescript
interface KpiData {
  occupancyPercent: number
  adr: number          // Average Daily Rate
  revpar: number       // Revenue Per Available Room
  todayCheckIns: number
  todayCheckOuts: number
  openFolios: number
  freeRooms: number
  totalRooms: number
  occupiedRooms: number
  dirtyRooms: number
  oooRooms: number
}
```

Svaka KPI kartica:
- Ikona (levo)
- Label (npr. "Popunjenost")
- Vrijednost (veliki broj)
- Trend strelica (↑ ili ↓) sa procentom
- Boja: primary za occupancy, success za check-in, warning za dirty, error za OOO

```typescript
function KpiCard({ icon: Icon, label, value, trend, color }: KpiCardProps) {
  return (
    <div className="rounded-xl bg-surface p-4 shadow-sm border border-border">
      <div className="flex items-center gap-3">
        <div className={`rounded-lg bg-${color}-100 p-2 text-${color}-600`}>
          <Icon className="h-5 w-5" />
        </div>
        <div className="flex-1">
          <p className="text-sm text-text-secondary">{label}</p>
          <p className="text-2xl font-semibold text-text">{value}</p>
        </div>
        {trend && (
          <span className={`text-sm ${trend.direction === 'up' ? 'text-success' : 'text-error'}`}>
            {trend.direction === 'up' ? '↑' : '↓'} {trend.percent}%
          </span>
        )}
      </div>
    </div>
  )
}
```

### 3.3 Chart.js integracija
Lokacija: `frontend/src/components/dashboard/OccupancyChart.tsx`

```typescript
'use client'

import { Line } from 'react-chartjs-2'
import {
  Chart as ChartJS,
  CategoryScale,
  LinearScale,
  PointElement,
  LineElement,
  Title,
  Tooltip,
  Legend,
  Filler,
} from 'chart.js'

ChartJS.register(
  CategoryScale, LinearScale, PointElement, LineElement,
  Title, Tooltip, Legend, Filler
)

export function OccupancyChart({ data }: { data: { date: string; occupancy: number; revenue: number }[] }) {
  return (
    <Line
      data={{
        labels: data.map(d => d.date),
        datasets: [
          {
            label: 'Popunjenost (%)',
            data: data.map(d => d.occupancy),
            borderColor: 'hsl(214 90% 42%)',
            backgroundColor: 'hsl(214 90% 42% / 0.1)',
            fill: true,
            tension: 0.3,
          },
          {
            label: 'Prihod (€)',
            data: data.map(d => d.revenue),
            borderColor: 'hsl(142 71% 45%)',
            backgroundColor: 'hsl(142 71% 45% / 0.1)',
            fill: true,
            tension: 0.3,
            yAxisID: 'y1',
          },
        ],
      }}
      options={{
        responsive: true,
        interaction: { intersect: false, mode: 'index' },
      }}
    />
  )
}
```

### 3.4 DashboardService
Lokacija: `frontend/src/lib/dashboard/dashboard-service.ts`

```typescript
export const dashboardService = {
  async getKpi(): Promise<KpiData> {
    const response = await apiClient.get('/dashboard/kpi')
    return response.data
  },
  async getOccupancyTrend(days: number = 30) {
    const response = await apiClient.get(`/dashboard/occupancy-trend?days=${days}`)
    return response.data
  },
  async getRecentBookings(limit: number = 10) {
    const response = await apiClient.get(`/dashboard/recent-bookings?limit=${limit}`)
    return response.data
  },
  async getUpcomingCheckIns(days: number = 7) {
    const response = await apiClient.get(`/dashboard/upcoming-checkins?days=${days}`)
    return response.data
  },
}
```
