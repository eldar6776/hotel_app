### 3.1 Room status state machine
```
                    ┌──────────┐
                    │   Free   │
                    └────┬─────┘
                         │
                    ┌────▼─────┐
                    │ Reserved │
                    └────┬─────┘
                         │ Check-in
                    ┌────▼──────┐
                    │ Occupied  │
                    └────┬──────┘
                         │ Check-out
                    ┌────▼─────┐
                    │  Dirty   │
                    └────┬─────┘
                         │ Cleaning
                    ┌────▼─────┐
                    │   Free   │
                    └──────────┘
```

Dozvoljene tranzicije:
- Free → Reserved (booking kreiran)
- Free → Occupied (walk-in check-in)
- Reserved → Occupied (check-in)
- Reserved → Free (booking cancelled)
- Occupied → Dirty (check-out)
- Dirty → Free (ciscenje zavrseno)
- Any → OutOfOrder (kvar)
- Any → OutOfService (renovacija)
- OutOfOrder → Free (popravka zavrsena)
- OutOfService → Free (renovacija zavrsena)

### 3.2 SignalR Hub
Lokacija: `backend/src/HotelPro.Api/Hubs/RoomStatusHub.cs`

```csharp
[Authorize]
public class RoomStatusHub : Hub
{
    private readonly IRoomService _roomService;

    public async Task JoinHotelGroup(string hotelCode)
    {
        await Groups.AddToGroupAsync(Context.ConnectionId, $"hotel_{hotelCode}");
    }

    public async Task LeaveHotelGroup(string hotelCode)
    {
        await Groups.RemoveFromGroupAsync(Context.ConnectionId, $"hotel_{hotelCode}");
    }

    public override async Task OnConnectedAsync()
    {
        // Automatski join u hotel grupu na osnovu JWT claim-a
        var hotelId = Context.User?.FindFirst("hotelId")?.Value;
        if (hotelId != null)
            await Groups.AddToGroupAsync(Context.ConnectionId, $"hotel_{hotelId}");
        await base.OnConnectedAsync();
    }
}
```

```typescript
import * as signalR from '@microsoft/signalr'

class RoomHubService {
  private connection: signalR.HubConnection | null = null

  async connect(hotelCode: string) {
    this.connection = new signalR.HubConnectionBuilder()
      .withUrl(`${process.env.NEXT_PUBLIC_API_URL}/hubs/room-status`, {
        accessTokenFactory: () => tokenStorage.getAccessToken() ?? ''
      })
      .withAutomaticReconnect([0, 2000, 5000, 10000, 30000])
      .build()

    this.connection.on('RoomStatusChanged', (data: RoomStatusChange) => {
      this.onStatusChange?.callbacks.forEach(cb => cb(data))
    })

    await this.connection.start()
    await this.connection.invoke('JoinHotelGroup', hotelCode)
  }

  onStatusChange(callback: (data: RoomStatusChange) => void) {
    // add callback
  }

  async disconnect() {
    await this.connection?.stop()
  }
}
```

### 3.7 Validacija state machine-a
```csharp
public static class RoomStatusTransitions
{
    private static readonly Dictionary<RoomStatus, HashSet<RoomStatus>> AllowedTransitions = new()
    {
        [RoomStatus.Free] = new() { RoomStatus.Reserved, RoomStatus.Occupied, RoomStatus.OutOfOrder, RoomStatus.OutOfService },
        [RoomStatus.Reserved] = new() { RoomStatus.Occupied, RoomStatus.Free, RoomStatus.OutOfOrder },
        [RoomStatus.Occupied] = new() { RoomStatus.Dirty, RoomStatus.OutOfOrder },
        [RoomStatus.Dirty] = new() { RoomStatus.Free, RoomStatus.OutOfOrder },
        [RoomStatus.OutOfOrder] = new() { RoomStatus.Free, RoomStatus.OutOfService },
        [RoomStatus.OutOfService] = new() { RoomStatus.Free },
    };

    public static bool IsValid(RoomStatus from, RoomStatus to)
        => AllowedTransitions.TryGetValue(from, out var targets) && targets.Contains(to);
}
```

- State machine validacija je na backendu (frontend ne moze poslati ilegalnu tranziciju)
- Reconnect interval: 0s, 2s, 5s, 10s, 30s (eksponencijalni backoff)
- Notifikacije su local (ne salju se email/SMS)
- Hub je zastiticen sa `[Authorize]`
